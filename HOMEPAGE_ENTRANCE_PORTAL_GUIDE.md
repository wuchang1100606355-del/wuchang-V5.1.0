在 `web_ui.py` 中添加：

```python
@app.route('/')
def index():
    index_path = os.path.join(os.path.dirname(__file__), 'index.html')
    if os.path.exists(index_path):
        with open(index_path, 'r', encoding='utf-8') as f:
            return f.read()
    else:
        return "首頁文件未找到", 404

@app.route('/wuchang-community')
```
